  }

# 简单的递归

function sun(n){
	if(n===1){
		return 1
	}else{
		return n+sun(n-1)
	}
}
sun(5) // 1+2+3+4+5 ==> 15

# 函数柯里化

function fn(template){
	return function(data){
    	return template.replace(`{name}`,data.name)
	}
}
var t=fn('<div>hi, I am {name}</div>')
t({name:"weibo"}) //"<div>hi, I am weibo</div>"
t({name:"jack"})  //"<div>hi, I am jack</div>"
